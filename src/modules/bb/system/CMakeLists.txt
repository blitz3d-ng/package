bb_start_module(system)
set(DEPENDS_ON bb.runtime)
set(SOURCES system.cpp system.h)
if(BB_MACOS)
  set(SOURCES ${SOURCES} system.macos.m)
  set(SYSTEM_LIBS "-framework Cocoa")
elseif(BB_IOS)
  set(SOURCES ${SOURCES} system.ios.m)
  set(SYSTEM_LIBS "-framework CoreGraphics" "-framework UIKit")
elseif(BB_NDK)
  set(SOURCES ${SOURCES} system.ndk.cpp)
endif()
bb_end_module()
